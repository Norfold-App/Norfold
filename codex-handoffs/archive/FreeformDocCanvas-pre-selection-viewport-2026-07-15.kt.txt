                                                onDragCancel = { localPlacements.remove(block.id) },

                                                    height = (current.height + dragAmount.y.toDp().value).coerceAtLeast(64f),
                                                )
                                            }
                                        },
                                )
                            }
                        }
                    }
                }
            }
        }
    }
